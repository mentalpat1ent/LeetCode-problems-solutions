12. Integer to Roman

Python Solution 1
```
class Solution:
    def intToRoman(self, num: int) -> str:

        cs = ('M', 'CM', 'D', 'CD', 'C', 'XC', 'L', 'XL', 'X', 'IX', 'V', 'IV', 'I')
        vs = (1000, 900, 500, 400, 100, 90, 50, 40, 10, 9, 5, 4, 1)

        roman = []

        for c, v in zip(cs, vs):
            
            while num >= v:
                num -= v
                roman.append(c)

        return ''.join(roman)
```

Python Solution 2
```
class Solution:
    def intToRoman(self, num: int) -> str:
        
        digits = list(str(num))
        n = len(digits) - 1
        nums = []
        roman = ''

        for digit in digits:

            nums.append(int(digit) * 10 ** n)
            n -= 1
        
        n = len(nums)

        for x in nums:

            x = int(str(x)[0])
        
            if n == 4:
                roman += 'M' * x

            elif n == 3:
                if x == 4:
                    roman += 'CD'
                elif x == 9:
                    roman += 'CM'
                elif x >= 5:
                    roman += 'D'
                    roman += 'C' * (x - 5)
                elif x < 5:
                    roman += 'C' * x

            
            elif n == 2:
                if x == 4:
                    roman += 'XL'
                elif x == 9:
                    roman += 'XC'
                elif x >= 5:
                    roman += 'L'
                    roman += 'X' * (x - 5)
                elif x < 5:
                    roman += 'X' * x
            
            else:
                if x == 4:
                    roman += 'IV'
                elif x == 9:
                    roman += 'IX'
                elif x >= 5:
                    roman += 'V'
                    roman += 'I' * (x - 5)
                elif x < 5:
                    roman += 'I' * x

            n -= 1
        
        return roman
                
```
